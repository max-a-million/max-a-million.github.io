				System.out.println(gcd(60, 24));

			public static int gcd(int x, int y) {
						
				int r = 0;
						
				do {
					r = x % y;
					x = y;
					y = r;
				  
				} while (r != 0);
						
				return x;
			}
		}

		/* Version 2 (iterative) */
		/*************************/
		
		public class Main
		{
			public static void main(String[] args)
			{
				System.out.println(gcd(60, 23));
			}

			public static int gcd(int x, int y) {

				int r = Math.min(x, y);

				while (true)
				  if (x % r == 0 && y % r == 0) 
					return r;
				  else
					r = r - 1;
			}
		}
